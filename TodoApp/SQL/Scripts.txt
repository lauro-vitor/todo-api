CREATE TABLE Todo(
  [TodoId] INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
  [Name] VARCHAR(255) NOT NULL,
  [IsDone] BIT NOT NULL DEFAULT 0
);
CREATE TABLE [User](
	[UserId] INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	[Name] VARCHAR(255) NOT NULL,
	[Login] VARCHAR(255) NOT NULL,
	[Password] VARCHAR(255) NOT NULL,
);
